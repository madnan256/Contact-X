2. DATABASE STRUCTURE (STRICTLY BASED ON BRD) 
Create the following tables exactly: 
//TABLE: users 
users{ 
id: uuid PRIMARY KEY DEFAULT gen_random_uuid() username (unique, required) 
passwordHash: text 
role: enum 
isActive: boolean 
createdAt: timestamp 
updatedAt: timestamp 
} 
// TABLE: contacts  
contacts {  
   id: uuid PRIMARY KEY DEFAULT gen_random_uuid()  
   prefix: text  
   firstName: text NOT NULL  
   middleName: text  
   lastName: text NOT NULL  
   suffix: text  
   prefixAr: text  
   firstNameAr: text  
   middleNameAr: text  
   lastNameAr: text  
   suffixAr: text  
   gender: text  // 'male' | 'female'  
   dateOfBirth: text  
   nationality: text  
   nationalId: text  
   contactType: text NOT NULL DEFAULT 'citizen'  // 'citizen'|'employee'|'external'|'vip'  
   currentPosition: text  
   currentEntityId: uuid REFERENCES entities(id)  
   currentEntityName: text  
   emails: jsonb DEFAULT '[]'  
   phones: jsonb DEFAULT '[]'  
   addresses: jsonb DEFAULT '[]'  
   classifications: jsonb DEFAULT '[]'  
   profileCompleteness: integer DEFAULT 0  
   isActive: boolean DEFAULT true  
   createdAt: timestamp DEFAULT now()  
   updatedAt: timestamp DEFAULT now()  
}  
   
// TABLE: entities  
entities {  
   id: uuid PRIMARY KEY DEFAULT gen_random_uuid()  
   nameEn: text NOT NULL  
   nameAr: text  
   entityType: text NOT NULL  // 'public'|'semi-government'|'private'|'international'|'ngo'  
   country: text  
   sector: text  
   registrationId: text  
   parentEntityId: uuid REFERENCES entities(id)  
   addresses: jsonb DEFAULT '[]'  
   contactPoints: jsonb DEFAULT '[]'  
   profileCompleteness: integer DEFAULT 0  
   isActive: boolean DEFAULT true  
   createdAt: timestamp DEFAULT now()  
   updatedAt: timestamp DEFAULT now()  
}  
   
// TABLE: contact_entity_relations  
contact_entity_relations {  
   id: uuid PRIMARY KEY DEFAULT gen_random_uuid()  
   contactId: uuid NOT NULL REFERENCES contacts(id) ON DELETE CASCADE  
   entityId: uuid NOT NULL REFERENCES entities(id) ON DELETE CASCADE  
   role: text NOT NULL  
   isPrimary: boolean DEFAULT false  
   isActive: boolean DEFAULT true  
   startDate: text  
   endDate: text  
   createdAt: timestamp DEFAULT now()  
}  
   
// TABLE: audit_logs  
audit_logs {  
   id: uuid PRIMARY KEY DEFAULT gen_random_uuid()  
   entityType: text NOT NULL  // 'contact' | 'entity'  
   entityId: uuid NOT NULL  
   action: text NOT NULL  // 'create' | 'update' | 'delete' | 'merge'  
   changes: jsonb  
   performedBy: uuid  
   createdAt: timestamp DEFAULT now()  
}  
   
// TABLE: duplicate_candidates  
duplicate_candidates {  
   id: uuid PRIMARY KEY DEFAULT gen_random_uuid()  
   entityType: text NOT NULL  // 'contact' | 'entity'  
   record1Id: uuid NOT NULL  
   record2Id: uuid NOT NULL  
   matchScore: integer NOT NULL  // 0-100  
   matchReasons: jsonb DEFAULT '[]'  
   status: text DEFAULT 'pending'  // 'pending' | 'merged' | 'dismissed'  
   createdAt: timestamp DEFAULT now()  
   resolvedAt: timestamp  
   resolvedBy: uuid  
}  
   
// TABLE: kpi_snapshots  
kpi_snapshots {  
   id: uuid PRIMARY KEY DEFAULT gen_random_uuid()  
   kpiName: text NOT NULL  
   kpiValue: numeric NOT NULL  
   kpiDetails: jsonb  
   snapshotDate: date DEFAULT now()  
}
